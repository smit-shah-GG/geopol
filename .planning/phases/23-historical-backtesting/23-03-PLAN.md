    /** A single backtest evaluation run. */
    export interface BacktestRun {
      id: string;
      label: string;
      description: string | null;
      window_size_days: number;
      slide_step_days: number;
      min_predictions_per_window: number;
      checkpoints_json: Record<string, string>;
      status: 'pending' | 'running' | 'completed' | 'cancelled' | 'failed';
      started_at: string | null;
      completed_at: string | null;
      total_windows: number;
      completed_windows: number;
      total_predictions: number;

      vs_polymarket_record_json: Record<string, number> | null;
      error_message: string | null;
      created_at: string;
    }

    /** Per-window evaluation result for a backtest run. */
    export interface BacktestResult {
      id: number;
      run_id: string;
      window_start: string;
      window_end: string;
      prediction_start: string;
      prediction_end: string;
      checkpoint_name: string;
      num_predictions: number;
      brier_score: number | null;
      mrr: number | null;
      hits_at_1: number | null;
      hits_at_10: number | null;
      calibration_bins_json: {
        bins: number[];
        predicted_avg: (number | null)[];
        observed_freq: (number | null)[];
        counts: number[];
      } | null;
      prediction_details_json: Array<{
        prediction_id: string;
        question: string;
        predicted_prob: number;
        outcome: number;
        brier: number;
      }> | null;
      polymarket_brier: number | null;
      geopol_vs_pm_wins: number | null;
      pm_vs_geopol_wins: number | null;
      weight_snapshot_json: Record<string, number> | null;
      created_at: string;
    }

    /** Drill-down response: run metadata + all window results. */
    export interface BacktestRunDetail {
      run: BacktestRun;
      results: BacktestResult[];
    }

    /** Available model checkpoint info. */
    export interface CheckpointInfo {
      name: string;
      model_type: 'tirgn' | 'regcn';
      path: string;
      metrics: Record<string, number> | null;
      created_at: string | null;
    }

    /** Payload for starting a new backtest run. */
    export interface StartBacktestRequest {
      label: string;
      description?: string;
      window_size_days?: number;
      slide_step_days?: number;
      min_predictions_per_window?: number;
      checkpoints: Record<string, string>;
    }

    Create BacktestingPanel.ts implementing the AdminPanel interface (mount/destroy lifecycle). Follow AccuracyPanel.ts as the structural reference.

    **Panel Structure:**

    The panel has two views: (A) run list view (default) and (B) drill-down view (when a run is selected).

    **View A: Run List**

    1. "Start New Run" button at the top that opens an inline form:
       - Label input (required, text)
       - Description textarea (optional)
       - Checkpoint selection: fetch available checkpoints via `client.getCheckpoints()`, render as a list of checkboxes grouped by model_type. At least one must be selected.
       - Window parameters: pre-filled with defaults (14 days, 7 step, 3 min) -- editable number inputs.
       - "Estimated API calls: ~N" warning text (rough estimate: total_predictions_in_range / windows * checkpoints). Can be a static note: "Each prediction burns one Gemini API call."
       - "Start Backtest" submit button. On click: call `client.startBacktestRun(config)`, collapse the form, refresh the run list.

    2. Run list table with columns: Label, Status (with color-coded badge), Checkpoints (comma-separated model types), Windows (completed/total), Brier, MRR, Created, Duration.
       - Status badges: pending=gray, running=blue (with pulsing animation), completed=green, cancelled=yellow, failed=red.
       - Running runs show a progress indicator: "3/8 windows" with a thin progress bar.
       - Click on any row opens the drill-down view (View B).
       - Cancel button for running/pending runs (inline in the row).

    3. Auto-refresh every 10s (faster than AccuracyPanel's 30s because running backtests need responsive progress updates).

    **View B: Drill-Down**

    1. Back button ("<- All Runs") to return to the run list.
    2. Run metadata header: label, description, status badge, started_at, completed_at, window config, checkpoints used.
    3. Four summary stat cards (progressive disclosure -- click to expand chart below):
       - **Brier Score**: aggregate_brier value, with small trend indicator
       - **Calibration**: mean absolute calibration error (computed client-side from bins)
       - **Hit Rate**: total correct / total predictions
       - **vs Polymarket**: geopol_wins / polymarket_wins / draws from vs_polymarket_record_json
    4. Expanded chart sections (rendered on card click, collapsed by default):

       **a. Brier Score Curves (d3 SVG):**
       - X axis: time (window_end dates), Y axis: Brier score (0 to max)
       - One line per checkpoint_name (color-coded)
       - Horizontal dashed reference line at Brier = 0.25 (random baseline)
       - Dots at each data point with tooltip showing window details on hover
       - Use `d3.scaleTime()` for X, `d3.scaleLinear()` for Y, `d3.line()` for curves
       - SVG viewBox for responsive sizing
       - Colors: use CSS variables (--accent, --text-secondary, or define chart-specific colors for multi-checkpoint comparison)

       **b. Calibration Reliability Diagram (d3 SVG):**
       - X axis: predicted probability (0 to 1), Y axis: observed frequency (0 to 1)
       - 45-degree diagonal reference line (perfect calibration)
       - Calibration curve from aggregated bins across all windows (merge bins client-side)
       - Bar chart (faint, behind curve) showing prediction count per bin
       - Use `d3.scaleLinear()` for both axes

       **c. Hit Rate / Accuracy:**
       - Simple stat display: total predictions, correct predictions, hit rate percentage
       - If multiple checkpoints: grouped bar chart via `d3.scaleBand()` showing per-checkpoint hit rate

       **d. Geopol vs Polymarket:**
       - Head-to-head summary: wins, losses, draws
       - If polymarket_brier data exists in results: overlay Geopol Brier and PM Brier curves on same time chart
       - Highlight the delta (areas where Geopol outperforms in green, underperforms in red)

    5. Export buttons at bottom of drill-down:
       - "Export CSV" button: calls `client.exportBacktestRun(runId, 'csv')`, triggers browser download
       - "Export JSON" button: calls `client.exportBacktestRun(runId, 'json')`, triggers browser download
       - Download trigger: create a temporary `<a>` with `URL.createObjectURL(blob)` and programmatic click
